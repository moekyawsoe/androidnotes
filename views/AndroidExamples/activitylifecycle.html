MainActivity.java

package dev.mks.codeplay;

import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.os.Bundle;
import android.os.PersistableBundle;
import android.util.Log;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;

public class MainActivity extends AppCompatActivity {
    Button btnActivity1;
    EditText etData;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        etData = findViewById(R.id.et_name);
    }

    @Override
    protected void onStart() {
        super.onStart();
        Log.i("onStart","onStart is starting...");
    }

    @Override
    protected void onStop() {
        super.onStop();
        Log.i("onStop","onStop is starting...");
    }

    @Override
    protected void onResume() {
        super.onResume();
        Log.i("onResume","onResume is starting...");
    }

    @Override
    protected void onRestart() {
        super.onRestart();
        Log.i("onRestart","onRestart is starting...");
    }

    @Override
    protected void onPause() {
        super.onPause();
        Log.i("onPause","onPause is starting...");
    }

    @Override
    protected void onDestroy() {
        super.onDestroy();
        Log.i("onDestroy","onDestroy is starting...");
    }

    @Override
    public void onSaveInstanceState(@NonNull Bundle outState, @NonNull PersistableBundle outPersistentState) {
        super.onSaveInstanceState(outState, outPersistentState);
        Log.i("onSaveInstanceState","onSaveInstanceState is starting...");
        outState.putString("name", String.valueOf(etData.getText()));
    }

    @Override
    protected void onRestoreInstanceState(@NonNull Bundle savedInstanceState) {
        super.onRestoreInstanceState(savedInstanceState);
        Log.i("onRestoreInstanceState","onRestoreInstanceState is starting...");
        if (savedInstanceState != null){
            String saved_name = savedInstanceState.getString("name");
            if (saved_name != null){
                etData.setText(saved_name);
            }
        }
    }
}
